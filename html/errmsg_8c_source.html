45 static void
46 do_err (int errnoflag, int error, const char *fmt, va_list ap)
47 {
48  char buf[MAXLINE];
49 
50  /* the '-1' is so we save room for terminating newline */
51  vsnprintf (buf, MAXLINE-1, fmt, ap);
52  if (errnoflag) /* we want the system error */
53  snprintf (buf+strlen(buf), MAXLINE-strlen (buf)-1, ": %s",
54  strerror (error));
55 
56  strcat (buf, "\n");
57 
58  fflush (stdout); /* in case stdio and stderr are the same */
59  fputs (buf, stderr);
60  fflush (NULL); /* all stdio output */
61 }

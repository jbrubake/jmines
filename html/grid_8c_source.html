131 new_grid (int height, int width, int nmines)
132 {
133  int i, y, x;
134  Cell *cellptr;
135 
136  Grid *grid = malloc (sizeof (Grid));
137  if (grid == NULL)
138  goto fail_alloc;
139 
140  /* clamp dimensions */
141  if (height > MAX_Y) height = MAX_Y;
142  if (width > MAX_X) width = MAX_X;
143 
144  /* create the cell array */
145  grid->cells = malloc (height * width * sizeof (Cell));
146  if (grid->cells == NULL)
147  {
148  free (grid);
149  goto fail_alloc;
150  }
151 
152  TRACE ((DBG_STD, "Created grid (0x%x)\n", grid));
153 
154  grid->height = height;
155  grid->width = width;
156  grid->nmines = nmines;
157  grid->nfound = 0; /* haven't found any mines yet */
158  grid->nflags = 0; /* no flags placed yet */
159 
160  /* start empty and covered */
161  cellptr = grid->cells;
162  for (i = 0; i < height * width; i++)
163  {
164  cellptr->contents = EMPTY;
165  cellptr->marker = NONE;
166  cellptr++;
167  }
168 
169  TRACE ((DBG_STD, "Created cells (0x%x)\n", grid->cells));
170 
171  /* boring seed */
172  srand ((unsigned)time (NULL));
173 
174  /* randomly place the mines */
175  for (i = 0; i < nmines;)
176  {
177  y = rand_limit (height-1);
178  x = rand_limit (width-1);
179 
180  cellptr = get_cell (grid, y, x);
181  if (cellptr->contents != MINE)
182  {
183  cellptr->contents = MINE;
184  i++; /* one less mine to place */
185 
186  /* the following creates the numbers correctly
187  *
188  * depends on ONE, TWO...EIGHT being defined
189  * as 1, 2...8 respectively
190  */
191  if ((cellptr = get_cell (grid, y-1, x-1)) != NULL &&
192  cellptr->contents != MINE)
193  cellptr->contents++;
194  if ((cellptr = get_cell (grid, y-1, x )) != NULL &&
195  cellptr->contents != MINE)
196  cellptr->contents++;
197  if ((cellptr = get_cell (grid, y-1, x+1)) != NULL &&
198  cellptr->contents != MINE)
199  cellptr->contents++;
200  if ((cellptr = get_cell (grid, y , x-1)) != NULL &&
201  cellptr->contents != MINE)
202  cellptr->contents++;
203  if ((cellptr = get_cell (grid, y , x+1)) != NULL &&
204  cellptr->contents != MINE)
205  cellptr->contents++;
206  if ((cellptr = get_cell (grid, y+1, x-1)) != NULL &&
207  cellptr->contents != MINE)
208  cellptr->contents++;
209  if ((cellptr = get_cell (grid, y+1, x )) != NULL &&
210  cellptr->contents != MINE)
211  cellptr->contents++;

213  cellptr->contents != MINE)

243 int
244 cell_reveal (Grid *grid, int y, int x)
245 {
246  Cell *cellptr;
247 
248  cellptr = get_cell (grid, y, x);
249  if (cellptr == NULL) return 0;
250 
251  /* do nothing if we try to reveal an already revealed cell */
252  if (cellptr->marker == REVEAL)
253  return 0;
254  else if (cellptr->contents == MINE)
255  return -1; /* oops! */
256  else if (cellptr->contents != EMPTY)
257  {
258  cellptr->marker = REVEAL; /* reveal non-empty, non-mine cells */
259  return 0;
260  }
261  else
262  {
263  /* cell is empty so, reveal it and all adjacent cells */
264  cellptr->marker = REVEAL;
265  cell_reveal (grid, y-1, x-1);
266  cell_reveal (grid, y-1, x );
267  cell_reveal (grid, y-1, x+1);
268  cell_reveal (grid, y , x-1);
269  cell_reveal (grid, y , x+1);
270  cell_reveal (grid, y+1, x-1);
271  cell_reveal (grid, y+1, x );
272  cell_reveal (grid, y+1, x+1);
273  }
274 
275  return 0;
276 }

289 cell_flag (Grid *grid, int y, int x)
290 {
291  /* placed too many flags */
292  if (grid->nflags == grid->nmines)
293  return -1;
294 
295  Cell *cellptr = get_cell (grid, y, x);
296  if (cellptr == NULL) return 1;
297 
298  cell_mark (cellptr, FLAG);
299 
300  grid->nflags++;
301
